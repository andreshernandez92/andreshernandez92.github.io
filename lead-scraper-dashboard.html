🐍 The Python Pipeline Code

The full script is 3 modules chained together. Run python pipeline.py and the dashboard updates automatically:

python 
# pipeline.py — Full Lead Scraping & Dashboard Pipeline
# Target: scrapethissite.com (publicly authorized sandbox)
# Output: Google Sheets → Looker Studio Dashboard

import requests
from bs4 import BeautifulSoup
import pandas as pd
import gspread
from google.oauth2.service_account import Credentials
import time, random

# ── MODULE 1: SCRAPER ────────────────────────────────────────
def scrape_leads():
    BASE_URL = "https://www.scrapethissite.com/pages/simple/"
    headers = {"User-Agent": "Mozilla/5.0 (educational portfolio project)"}

    response = requests.get(BASE_URL, headers=headers)
    soup = BeautifulSoup(response.text, "html.parser")

    records = []
    for country in soup.find_all("div", class_="col-md-4 country"):
        name       = country.find("country-name")
        capital    = country.find("country-capital")
        population = country.find("country-population")
        area       = country.find("country-area")

        if all([name, capital, population]):
            records.append({
                "Business Name":  name.text.strip(),
                "Market / City":  capital.text.strip(),
                "Market Size":    int(population.text.strip()),
                "Territory (km²)": float(area.text.strip()) if area else 0.0,
            })
        time.sleep(random.uniform(0.1, 0.3))  # polite delay

    return pd.DataFrame(records)

# ── MODULE 2: CLEANER & ENRICHER ─────────────────────────────
def clean_and_score(df):
    # Remove duplicates and missing values
    df = df.drop_duplicates().dropna(subset=["Business Name", "Market / City"])

    # Normalize text fields
    df["Business Name"] = df["Business Name"].str.strip().str.title()
    df["Market / City"]  = df["Market / City"].str.strip().str.title()

    # Calculated enrichment column: Population Density
    df["Density (pop/km²)"] = (df["Market Size"] / df["Territory (km²)"].replace(0, pd.NA)).round(1)

    # Lead scoring based on market size
    def score(pop):
        if   pop > 50_000_000: return "🔥 Priority A"
        elif pop > 10_000_000: return "✅ Priority B"
        elif pop > 1_000_000:  return "📬 Priority C"
        else:                  return "❄️ Low Priority"

    df["Lead Score"] = df["Market Size"].apply(score)
    return df

# ── MODULE 3: GOOGLE SHEETS SYNC ─────────────────────────────
def push_to_sheets(df, spreadsheet_name="Lead Intelligence Dashboard"):
    scopes = ["https://www.googleapis.com/auth/spreadsheets"]
    creds  = Credentials.from_service_account_file("credentials.json", scopes=scopes)
    client = gspread.authorize(creds)
    sheet  = client.open(spreadsheet_name).sheet1
    sheet.clear()
    sheet.append_row(df.columns.tolist())
    sheet.append_rows(df.fillna("").values.tolist())
    print(f"✅ {len(df)} leads synced to Google Sheets!")

# ── RUN ──────────────────────────────────────────────────────
if __name__ == "__main__":
    print("🕷️  Starting scrape...")
    raw_df   = scrape_leads()
    clean_df = clean_and_score(raw_df)
    clean_df.to_csv("leads_output.csv", index=False)
    push_to_sheets(clean_df)
    print("🚀 Pipeline complete. Dashboard is updated!")

🛠️ Setup Instructions

  • 1. Install dependencies: pip install requests beautifulsoup4 pandas gspread google-auth
  • 2. Enable Google Sheets API in your Google Cloud Console and create a service account. Download credentials.json.
  • 3. Create a Google Sheet named "Lead Intelligence Dashboard" and share it with your service account email (Editor access).
  • 4. Run the pipeline: python pipeline.py — the script scrapes, cleans, scores, and syncs everything in under 30 seconds.
  • 5. Connect Looker Studio to your Google Sheet. Build a bar chart (Lead Score distribution), a scoreboard (total leads), and a sortable table.
